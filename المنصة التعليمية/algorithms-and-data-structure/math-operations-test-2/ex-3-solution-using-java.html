تمارين وتطبيقاتالتحدي الثاني - حل التمرين الثالث بلغة جافا

المطلوب

أكتب برنامج يطلب من المستخدم إدخال ثلاث أرقام و خزنها في ثلاث متغيرات (a - b - c ), ثم يعرض له أصغر رقم تم إداخله.
مثال: إذا قام المستخدم بإدخال الأرقام 2, 7 و 5 فستكون النتيجة كالتالي.

Enter a: 2
Enter b: 7
Enter c: 5
The min number is: 2

الحل بلغة جافا

الطريقة الأولى لحل التمرين.

import java.util.Scanner;

public class Main {

    public static void main(String[] args) {
        
        Scanner input = new Scanner(System.in);
        int a, b, c, minimum;
        
        System.out.print("Enter a: ");
        a = input.nextInt();
        
        System.out.print("Enter b: ");
        b = input.nextInt();
        
        System.out.print("Enter c: ");
        c = input.nextInt();
        
        minimum = (a < b)? a: b;
        minimum = (minimum < c)? minimum: c;
        
        System.out.println("The min number is: " + minimum);
    
    }

}

الطريقة الثانية لحل التمرين و الحصول على نفس النتيجة.

import java.util.Scanner;

public class Main {

    public static void main(String[] args) {
        
        Scanner input = new Scanner(System.in);
        int a, b, c, minimum;
        
        System.out.print("Enter a: ");
        a = input.nextInt();
        
        System.out.print("Enter b: ");
        b = input.nextInt();
        
        System.out.print("Enter c: ");
        c = input.nextInt();
        
        if (a < b && a < c)
        {
            minimum = a;
        }
        else if (b < a && b < c)
        {
            minimum = b;
        }
        else
        {
            minimum = c;
        }
        
        System.out.println("The min number is: " + minimum);
    
    }

}

الطريقة الثالثة لحل التمرين و الحصول على نفس النتيجة.

import java.util.Scanner;

public class Main {

    public static void main(String[] args) {
        
        Scanner input = new Scanner(System.in);
        int a, b, c, minimum;
        
        System.out.print("Enter a: ");
        a = input.nextInt();
        
        System.out.print("Enter b: ");
        b = input.nextInt();
        
        System.out.print("Enter c: ");
        c = input.nextInt();
        
        if (a < b && a < c)
        {
            minimum = a;
        }
        else
        {
            minimum = b;
        }
        
        if (minimum > c)
        {
            minimum = c;
        }
        
        System.out.println("The min number is: " + minimum);
    
    }

}
